#!/bin/bash

file=$1
# echo $file
sigo=$(echo "$file" | grep 'k\[' )


a=${sigo%?};


if [ "$a" == "*k[]" ]; then
    echo Key Signature: C Major
elif [ "$a" == "*k[f#]" ]; then
    echo Key Signature: G Major
elif [ "$a" == "*k[f#c#]" ]; then
    echo Key Signature: D Major
elif [ "$a" == "*k[f#c#g#]" ]; then
    echo Key Signature: A Major
elif [ "$a" == "*k[f#c#g#d#]" ]; then
    echo Key Signature: E Major
elif [ "$a" == "*k[f#c#g#d#a#]" ]; then
    echo Key Signature: B Major
elif [ "$a" == "*k[f#c#g#d#a#e#]" ]; then
    echo Key Signature: F\# Major
elif [ "$a" == "*k[f#c#g#d#a#e#b#]" ]; then
    echo Key Signature: C\#  Major
elif [ "$a" == "*k[b-]" ]; then
    echo Key Signature: F Major
elif [ "$a" == "*k[b-e-]" ]; then
    echo Key Signature: Bb Major
elif [ "$a" == "*k[b-e-a-]" ]; then
    echo Key Signature: Eb Major
elif [ "$a" == "*k[b-e-a-d-]" ]; then
    echo Key Signature: Ab Major
elif [ "$a" == "*k[b-e-a-d-g-]" ]; then
    echo Key Signature: Db Major
elif [ "$a" == "*k[b-e-a-d-g-c-]" ]; then
    echo Key Signature: Gb Major
elif [ "$a" == "*k[b-e-a-d-g-c-f-]" ]; then
    echo Key Signature: Cb Major
fi
